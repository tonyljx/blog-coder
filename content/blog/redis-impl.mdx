# sds



数据结构

```c
typedef struct sdshdr {
  int len;
  int free;
  char buff[]
}
```

```mermaid
flowchart LR
  SDS["sdshdr"] --> Len["len: 已使用长度"]
  SDS --> Free["free: 预留空间"]
  SDS --> Buf["buf[]: 字节数组"]

  Len --> O1["获取长度 O(1)"]
  Free --> Prealloc["减少扩容次数"]
  Buf --> Data["真实字符串内容"]
```



好处

- 获取字符串长度是 `O(1)` 的，默认字符串长度c 语言函数没有内置，只能获取
- 避免重复分配内存：预留空间，避免多次执行分配内存的系统调用





# 链表



```c
typedef struct listNode {
  struct listNode *prev;
  struct listNode *next;
  void * val;
}
```



```c
typedef struct list {
  listNode * head;
  listNode * tail;
  unsigned long len;
}
```

```mermaid
flowchart LR
  List["list\nhead\ntail\nlen = 3"]
  A["listNode A\nprev = NULL\nnext -> B\nvalue = ..."]
  B["listNode B\nprev -> A\nnext -> C\nvalue = ..."]
  C["listNode C\nprev -> B\nnext = NULL\nvalue = ..."]
  NullL["NULL"]
  NullR["NULL"]

  List -- "head" --> A
  List -- "tail" --> C
  List -- "len" --> Len["3, O(1)"]

  NullL -. "prev" .-> A
  A -- "next" --> B
  B -- "next" --> C
  C -- "next" --> NullR
  C -. "prev" .-> B
  B -. "prev" .-> A
```



特点：

- 获取链表长度是 `O(1)`





# 字典

用一个数组配合链表来实现

```c
typedef struct dictht {
  dicEntry ** table;
  unsigned long size; // 大小, 用来计算表格大小
  unsigned long sizeMask; // 掩码 用来计算哈希位置
  unsigned long used;  // 计算使用的, 结合 used & size 来计算 是否需要 rehash
}
```



```c
// 字典的每个值的带下
typedef struct dictEntry {
	void * key; // 键
  val union { // 值是枚举类型, 可以是指针 也可以是整数
    void * val;
    uint64_tu64; // 有可能有错
    int64_ts64;
  }
  struct dictEntry * next
}
```

```mermaid
flowchart LR
  HT["dictht\n table\n size = 4\n sizemask = 3\n used = 2"]

  subgraph Table["dictEntry* table[4]"]
    direction TB
    B0["bucket[0]"]
    B1["bucket[1]"]
    B2["bucket[2]"]
    B3["bucket[3]"]
  end

  HT -- "table 指针" --> B0
  HT -- "hash(key) & sizemask" --> Index["定位 bucket 下标"]
  Index --> B3

  B0 --> Null0["NULL"]
  B1 --> Entry1["dictEntry\nkey: k1\nvalue: v1\nnext"]
  Entry1 --> Null1["NULL"]
  B2 --> Null2["NULL"]
  B3 --> Entry3A["dictEntry\nkey: k2\nvalue: v2\nnext"]
  Entry3A --> Entry3B["dictEntry\nkey: k3\nvalue: v3\nnext"]
  Entry3B --> Null3["NULL"]
```

```mermaid
flowchart LR
  subgraph OldTable["ht[0] 旧表"]
    OldBucket["bucket 链表"]
  end

  subgraph NewTable["ht[1] 新表"]
    NewBucket["bucket 链表"]
  end

  Trigger["used/size 到达阈值"] --> Alloc["分配 ht[1]"]
  Alloc --> NewTable
  Write["新增 key"] --> NewBucket
  Step["每次操作顺手迁移一部分 bucket"] --> Move["从 ht[0] 搬到 ht[1]"]
  OldBucket --> Move --> NewBucket
  Query["查询/更新 key"] --> Lookup["同时查 ht[0] 和 ht[1]"]
  Lookup --> OldTable
  Lookup --> NewTable
```



- 插入重复值：基于链表实现，因为没办法获取尾部节点，所以每次插入的值，会优先放到最前面
- rehash
  - 时机：
    - 服务器没有启动额外子进程【没有备份】， used >= size
    - 服务器启动了 额外子进程 ： used / size >= 配置的值，一般是 5
  - 动作：渐进式 rehash
    - 分配一个 `ht[1]` ：size 是 2 的 n 次方幂 >= used `*` 2
    - 新增写入：只会写入新的哈希数组
    - 存量写入：更新旧表，也会更新新表





# 跳表



```c
typedef struct zset {
	dict *dict;
  zskiplist *zsl;
} zset
```



```c
typedef struct zskiplistNode {
    robj *obj;
    double score;
    struct zskiplistNode *backward;
    struct zskiplistLevel {
        struct zskiplistNode *forward;
        unsigned int span;
    } level[];
} zskiplistNode;
```

```mermaid
flowchart LR
  ZSL["zskiplist\nheader\ntail\nlevel = 4\nlength = 3"]
  Header["header\nlevel[3]\nlevel[2]\nlevel[1]\nlevel[0]"]
  A["zskiplistNode A\nlevel[0]\nbackward = NULL\nscore = 1.0\nobj = o1"]
  B["zskiplistNode B\nlevel[1]\nlevel[0]\nbackward -> A\nscore = 2.0\nobj = o2"]
  C["zskiplistNode C\nlevel[3]\nlevel[2]\nlevel[1]\nlevel[0]\nbackward -> B\nscore = 3.0\nobj = o3"]
  Null["NULL"]

  ZSL -- "header" --> Header
  ZSL -- "tail" --> C

  Header -- "level[0].forward, span=1" --> A
  A -- "level[0].forward" --> B
  B -- "level[0].forward" --> C
  C -- "forward" --> Null

  Header -- "level[1].forward, span=2" --> B
  Header -- "level[3].forward, span=3" --> C

  C -. "backward" .-> B
  B -. "backward" .-> A
```





能力：dict 负责根据 member 找 score  `O(1)`，zsl 根据 score 来批量查询/更新 member `o(log n)`



```mermaid
flowchart LR
  H3["level 3: header"] -- "span=4" --> D3["D"]

  H2["level 2: header"] -- "span=2" --> B2["B"]
  B2 -- "span=2" --> D2["D"]

  H1["level 1: header"] -- "span=1" --> A1["A"]
  A1 -- "span=1" --> B1["B"]
  B1 -- "span=1" --> C1["C"]
  C1 -- "span=1" --> D1["D"]

  H0["level 0: header"] --> A0["A"]
  A0 --> B0["B"]
  B0 --> C0["C"]
  C0 --> D0["D"]
```



QA

- 层高如何确定：随机确定的，25% 的概率，大概率是底层，最高 32 层，越往上概率越小
- 如何保持平衡：通过随机层高来保持平衡
- span 跨度来计算排名：快速获取 rank
